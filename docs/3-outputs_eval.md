```python
def remove_broken_files():
    error_id = []
    for f in os.listdir('ref'):
        try:
            open('ref/' + f).read()
        except:
            error_id.append(f)
    for f in os.listdir('hyp'):
        try:
            open('hyp/' + f).read()
        except:
            error_id.append(f)
    error_set = set(error_id)
    for f in error_set:
        os.remove('ref/' + f)
        os.remove('hyp/' + f)
```
